    def save(obj: object, path: Path) -> None:
    """Write object to pickle file at given path."""
    if not path.parent.exists():
        path.parent.mkdir(parents=True, exist_ok=True)
    with Path.open(path, "wb") as f:
        pickle.dump(obj, f)
